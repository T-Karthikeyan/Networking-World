//<----------------------------------- RAIL FENCE CIPHER --------------------------------->

#include <stdio.h>
#include <string.h>

void main()
{
    int i, j, len, depth, zigzag;
    char pt[100], ct[10][100];
    printf("~~~~~~~~~~~ RAIL FENCE CIPHER ~~~~~~~~~~");
    printf("\n\nEnter the plain text : ");
    scanf("%s",pt);
    printf("Enter the depth : ");
    scanf("%d", &depth);
    len = strlen(pt);
    
    // FILL ALL THE BOXES OF GIVEN LENGTH WITH '.'
    for(i = 0; i < depth; i++)
        for(j = 0; j < len; j++)
            ct[i][j] = '.';
    
    // TRAVERSE ZIGZAG(UP-DOWN = 0, DOWN-UP = 1) AND PLACE CHARACTERS IN THEIR RESPECTIVE POSITIONS.
    for(j = 0, zigzag = 0; j < len; zigzag++)
    {
        if(zigzag % 2 == 0)
            for(i = 0; i < depth; i++, j++)
                ct[i][j] = pt[j];
        else
            for(i = depth - 2; i > 0; i--, j++)
                ct[i][j] = pt[j];
    }
    
    // DISPLAY THE ENCRYPTED TEXT - CHARACTERS IN ORDER EXCEPT '.'
    printf("Encrypted Text : ");
    for(i = 0; i < depth; i++)
        for(j = 0; j < len; j++)
            if(ct[i][j] != '.')
                printf("%c", ct[i][j]);
    printf("\n\n~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~");
}
